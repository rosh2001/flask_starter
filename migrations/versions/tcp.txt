import socket

#Create client socket
clientSocket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

#Establish destination address
serverName = '127.0.0.1'
serverPort = 12345

#Connect to server
clientSocket.connect((serverName, serverPort))

print(clientSocket.recv(1024).decode())

while True:
    #Send data for conversion
    data = input('Enter data to convert to uppercase: ')
    clientSocket.send(data.encode())

    print(clientSocket.recv(1024).decode())


import socket

#Create server socket
serverSocket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

#Establish server address
serverName = '127.0.0.1'
serverPort = 12345

#Bind server socket and address
serverSocket.bind((serverName, serverPort))

#Listen for incoming TCP connections
serverSocket.listen(1)
print('Listening for incoming connections ...')

while True:
    connectionSocket, clientAddress = serverSocket.accept()
    print('Client', clientAddress, 'wants to connect')
    connectionSocket.send('Thanks for connecting'.encode())

    while True:
        modData = connectionSocket.recv(1024).decode().upper()
        connectionSocket.send(modData.encode())

    connectionSocket.close()
    
    break
